CREATE TABLE CCEMPLEADO (
    ID           NUMBER(10, 0)         NOT NULL,
    CLAVE        NVARCHAR2(50)         NOT NULL,
    NOMBRECOMPLETO  NVARCHAR2(255)     NOT NULL,
    CONTRASENA   NVARCHAR2(50),
    ESACTIVO     NUMBER(1,0),
    CONSTRAINT PK_CCEMPLEADO PRIMARY KEY (ID)
);
CREATE SEQUENCE CCEMPLEADO_SEQ
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;
CREATE OR REPLACE TRIGGER CCEMPLEADO_ID_TRG
BEFORE INSERT ON CCEMPLEADO 
FOR EACH ROW
BEGIN
    IF :NEW.ID IS NULL THEN
        SELECT CCEMPLEADO_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
    END IF;
END;
/


COMMIT;
