CREATE TABLE CCROL (
    ID           NUMBER(10, 0)         NOT NULL,
    NOMBRE       NVARCHAR2(255)        NOT NULL,
    CONSTRAINT PK_CCROL PRIMARY KEY (ID)
);

CREATE SEQUENCE CCROL_SEQ
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;

CREATE OR REPLACE TRIGGER CCROL_ID_TRG
BEFORE INSERT ON CCROL 
FOR EACH ROW
BEGIN
    IF :NEW.ID IS NULL THEN
        SELECT CCROL_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
    END IF;
END;
/

INSERT INTO CCROL (ID, NOMBRE) VALUES (CCROL_SEQ.NEXTVAL, 'ADMINISTRADOR');


COMMIT;
