CREATE TABLE CCCONFIG (
    ID           NUMBER(10, 0)         NOT NULL,
    CLAVE        NVARCHAR2(50)         NOT NULL,
    DESCRIPCION  NVARCHAR2(255)     NOT NULL,
    CONSTRAINT PK_CCCONFIG PRIMARY KEY (ID)
);
CREATE SEQUENCE CCCONFIG_SEQ
    START WITH 1
    INCREMENT BY 1
    NOCACHE
    NOCYCLE;
CREATE OR REPLACE TRIGGER CCCONFIG_ID_TRG
BEFORE INSERT ON CCCONFIG 
FOR EACH ROW
BEGIN
    IF :NEW.ID IS NULL THEN
        SELECT CCCONFIG_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
    END IF;
END;
